model{
	
	lambda~dgamma(0.001,0.001)
	
	Psychological~dnorm(0,1)
	Physical~dnorm(0,1)
	
	outcome_options[1] <- Psychological
	outcome_options[2] <- Physical
	
	# Loop through studies (upper.group)
	for (s in 1:upper.group) {
		
		theta[s]~dnorm(0,1)
		
		# Loop through observations (lower.group)
		for (o in 1:lower.group[s]) {
			
			outcome[s,o] <-  outcome_options[outcomeOptionsMatrix[s,o]]
			eta[s,o] <- theta[s] + outcome[s,o]
			outcomesMatrix[s,o] ~ dnorm(eta[s,o],lambda)
			
		}
	}
}
